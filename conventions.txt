params
[
	["_mode", "", [""]],
	["_moduleParams", [], [[]]]
];
_moduleParams params
[
	["_logic", objNull, [objNull]],
	["_isActivated", true, [true]],
	["_isCuratorPlaced", true, [true]],
	["_params", [], []]
];

switch (_mode) do
{
	case "init":
	{
		["selection", ["OBJECT", "MAN"]] call Achilles_fnc_module_init;
	};
	case "postinit":
	{
		private _selectedUnits = _logic getVariable ["#selection", []];
		[
			localize "STR_AMAE_ACE_INJURY",
			[
				...
			]
		] call Achilles_fnc_module_openDialog;
	};
	case "confirmed":
	{
		deleteVehicle _logic;
		["Confirmed"] call Achilles_fnc_printZeusMessage;
		["Error"] call Achilles_fnc_printZeusError;
	};
	case "canceled":
	{
		deleteVehicle _logic;
	};
};
